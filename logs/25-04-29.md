```c
#include <stdio.h>
int main(int argc, char *argv[])
{
  int arr[2][3] = {1,2,3,4,5,6};
  int (*p)[3] = NULL;     // int 3개짜리 배열을 가리키는 포인터 선언 (*p[3]: int 타입을 가리키는 포인터 3개짜리 배열 선언) 
  p = arr;

  printf("%d\n", *(p[0] + 1) + *(p[1] + 2));                // 8 (2+6)
  printf("%d\n", *(*(p + 1) + 0) + *(*(p + 1) + 1));        // 9 (4+5)
  return 0;
}
